       ```typescript
       import { create } from "zustand";
       import { persist } from "zustand/middleware";

       interface SyncStore {
         // Delta links
         listsDeltaLink: string | null;
         tasksDeltaLinks: Record<string, string | null>; // keyed by list ID

         // Sync metadata
         lastSyncedAt: number | null;
         syncStatus: "idle" | "syncing" | "error" | "resyncing";

         // UI selection
         selectedListId: string | null;

         // Flagged emails list
         flaggedEmailsListId: string | null;

         // Actions
         setListsDeltaLink: (link: string | null) => void;
         setTasksDeltaLink: (listId: string, link: string | null) => void;
         setLastSyncedAt: (ts: number) => void;
         setSyncStatus: (s: SyncStore["syncStatus"]) => void;
         setSelectedListId: (id: string | null) => void;
         setFlaggedEmailsListId: (id: string | null) => void;
         clearTasksDeltaLink: (listId: string) => void;
         resetAllDeltaLinks: () => void;
       }

       export const useSyncStore = create<SyncStore>()(
         persist(
           (set) => ({
             listsDeltaLink: null,
             tasksDeltaLinks: {},
             lastSyncedAt: null,
             syncStatus: "idle",
             selectedListId: null,
             flaggedEmailsListId: null,

             setListsDeltaLink: (link) => set({ listsDeltaLink: link }),
             setTasksDeltaLink: (listId, link) =>
               set((state) => ({
                 tasksDeltaLinks: { ...state.tasksDeltaLinks, [listId]: link },
               })),
             setLastSyncedAt: (ts) => set({ lastSyncedAt: ts }),
             setSyncStatus: (s) => set({ syncStatus: s }),
             setSelectedListId: (id) => set({ selectedListId: id }),
             setFlaggedEmailsListId: (id) => set({ flaggedEmailsListId: id }),
             clearTasksDeltaLink: (listId) =>
               set((state) => ({
                 tasksDeltaLinks: { ...state.tasksDeltaLinks, [listId]: null },
               })),
             resetAllDeltaLinks: () =>
               set({
                 listsDeltaLink: null,
                 tasksDeltaLinks: {},
               }),
           }),
           {
             name: "ms-todo-sync-store",
             partialize: (state) => ({
               listsDeltaLink: state.listsDeltaLink,
               tasksDeltaLinks: state.tasksDeltaLinks,
               lastSyncedAt: state.lastSyncedAt,
               selectedListId: state.selectedListId,
               flaggedEmailsListId: state.flaggedEmailsListId,
             }),
           }
         )
       );
       ```

       Key design decisions:
       - `partialize` excludes `syncStatus` from persistence (it should start as "idle" on fresh load, not stuck in "syncing")
       - `flaggedEmailsListId` is persisted so we don't re-scan on every load (but `useTaskLists` will update it if lists change)
       - `tasksDeltaLinks` is a Record keyed by list ID â€” each list has its own independent delta token
       - `resetAllDeltaLinks` is for the 410 Gone full-reset scenario

    1. Create `src/features/task-lists/api/taskLists.api.ts` â€” raw API function for task list delta sync:
       ```typescript
       import { graphFetch } from "../../graph";
       import type { GraphPagedResponse } from "../../graph";
       import type { TodoTaskList } from "@microsoft/microsoft-graph-types";

       /**
        * Fetches all task lists via delta query. Paginates through ALL @odata.nextLink
        * pages before returning. Returns the accumulated lists AND the final deltaLink.
        *
        * CRITICAL: Never save a deltaLink until ALL nextLink pages are consumed.
        * The deltaLink only appears on the LAST page.
        */
       export async function fetchTaskListsDelta(
         getToken: () => Promise<string>,
         deltaLink: string | null
       ): Promise<{ lists: TodoTaskList[]; deltaLink: string | null }> {
         const url = deltaLink ?? "/me/todo/lists/delta";
         const allLists: TodoTaskList[] = [];
         let nextUrl: string | null = url;
         let finalDeltaLink: string | null = null;

         while (nextUrl) {
           const response = await graphFetch<GraphPagedResponse<TodoTaskList>>(
             nextUrl,
             getToken
           );
           allLists.push(...response.value);

           if (response["@odata.nextLink"]) {
             nextUrl = response["@odata.nextLink"];
           } else {
             finalDeltaLink = response["@odata.deltaLink"] ?? null;
             nextUrl = null;
           }
         }

         return { lists: allLists, deltaLink: finalDeltaLink };
       }
       ```

       Note: This uses `graphFetch` (single page) NOT `graphFetchAll`, because we need fine-grained control over the pagination loop to correctly capture the deltaLink from the last page only. The `graphFetchAll` utility from Phase 1 is designed for simple pagination; delta queries need the explicit loop to handle the nextLink/deltaLink distinction.

    2. Create `src/features/task-lists/hooks/useTaskLists.ts` â€” TanStack Query hook for task list delta sync:
       ```typescript
       import { useQuery, useQueryClient } from "@tanstack/react-query";
       import type { TodoTaskList } from "@microsoft/microsoft-graph-types";
       import { useGraphToken } from "../../auth";
       import { useSyncStore } from "../../../stores/sync.store";
       import { fetchTaskListsDelta } from "../api/taskLists.api";

       export function useTaskLists() {
         const { getToken } = useGraphToken();
         const queryClient = useQueryClient();

         return useQuery({
           queryKey: ["taskLists"],
           queryFn: async (): Promise<TodoTaskList[]> => {
             const store = useSyncStore.getState();
             store.setSyncStatus("syncing");

             try {
               const { lists, deltaLink } = await fetchTaskListsDelta(
                 getToken,
                 store.listsDeltaLink
               );

               // On initial sync (no prior deltaLink), replace the full cache.
               // On incremental sync (had a deltaLink), merge changes into existing cache.
               const existingLists =
                 queryClient.getQueryData<TodoTaskList[]>(["taskLists"]) ?? [];

               let mergedLists: TodoTaskList[];

               if (store.listsDeltaLink === null) {
                 // Initial sync â€” use full result
                 mergedLists = lists;
               } else {
                 // Incremental sync â€” merge delta into existing
                 const listMap = new Map(
                   existingLists.map((l) => [l.id, l])
                 );
                 for (const list of lists) {
                   if ((list as any)["@removed"]) {
                     listMap.delete(list.id);
                   } else {
                     const existing = listMap.get(list.id);
                     listMap.set(list.id, existing ? { ...existing, ...list } : list);
                   }
                 }
                 mergedLists = Array.from(listMap.values());
               }

               // Save the new deltaLink (per SYNC-01: store full URL, never parse)
               if (deltaLink) {
                 store.setListsDeltaLink(deltaLink);
               }

               // Detect flaggedEmails list (per LIST-02)
               const flaggedList = mergedLists.find(
                 (l) => l.wellknownListName === "flaggedEmails"
               );
               if (flaggedList?.id) {
                 store.setFlaggedEmailsListId(flaggedList.id);
               }

               // Auto-select first list if none selected
               if (!store.selectedListId && mergedLists.length > 0) {
                 store.setSelectedListId(mergedLists[0].id ?? null);
               }

               store.setSyncStatus("idle");
               store.setLastSyncedAt(Date.now());
               return mergedLists;
             } catch (error: any) {
               // Handle 410 Gone (per SYNC-03) â€” clear deltaLink, restart full sync
               if (error?.message?.includes("410") || error?.code === "GoneError") {
                 store.setListsDeltaLink(null);
                 store.setSyncStatus("resyncing");
                 // Retry with no deltaLink (full sync)
                 const { lists, deltaLink: newDeltaLink } =
                   await fetchTaskListsDelta(getToken, null);

                 if (newDeltaLink) {
                   store.setListsDeltaLink(newDeltaLink);
                 }

                 const flaggedList = lists.find(
                   (l) => l.wellknownListName === "flaggedEmails"
                 );
                 if (flaggedList?.id) {
                   store.setFlaggedEmailsListId(flaggedList.id);
                 }

                 if (!store.selectedListId && lists.length > 0) {
                   store.setSelectedListId(lists[0].id ?? null);
                 }

                 store.setSyncStatus("idle");
                 store.setLastSyncedAt(Date.now());
                 return lists;
               }

               store.setSyncStatus("error");
               throw error;
             }
           },
           // Initial fetch enabled only when we have a token provider
           enabled: !!getToken,
         });
       }
       ```

       Key implementation decisions:
       - `queryFn` reads the deltaLink from Zustand store (outside React, via `getState()`) â€” this is the recommended pattern for accessing Zustand inside TanStack Query queryFn
       - Delta merge logic: on incremental sync, merge changed lists into existing cache map; handle `@removed` for deleted lists
       - 410 Gone: clear deltaLink and immediately retry with full sync, set syncStatus to "resyncing"
       - Auto-selects first list if nothing selected (UX convenience)
       - `flaggedEmails` detection happens on every sync to keep it current
